The secret sauce

public function shouldHaveCalledAction(...$actions): void
{
	collect($actions)
		->flatten()
		->each(function(string $action) {
			$original = $this->app->make($action);

			$this->mock($action)
				->shouldReceive('__invoke')
				->atLeast()->once()
				->andReturnUsing(fn(...$args) => $original(...$args));
		});
}
